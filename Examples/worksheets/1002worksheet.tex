		\question Let $f(x)=x+2$, $g(x)=2x^2-1$.
    \begin{parts}
        \part Compute $f'(x)$ and $g'(x)$
	    \begin{solution}[2cm]
		    $f'(x)=1$ and $g'(x) = 4 x$
	    \end{solution}
        \part Compute $[f(x)g(x)]'$. How does it compare to $f'(x)g'(x)$?
	    \begin{solution}[2cm]
		    $f'(x)g'(x) = 4x$ and $(f(x)g(x))' = 1 (2x^2 -1) + 4x (x+2) = 6x^2 + 8x - 1$. 
	    \end{solution}
        \part Compute $[f(g(x))]'$. How does it compare to $f'(g'(x))$?
	    \begin{solution}[2cm]
		    $f'(g'(x)) = 1$ and $(f(g(x)))' = ((2x^2 - 1) + 2)' = 4 x$.
	    \end{solution}
	\part Compute $\frac{f'(x)}{g'(x)}$ and $(\frac{f(x)}{g(x)})'$.
	    \begin{solution}[2cm]
		    $\frac{f'(x)}{g'(x)} = \frac{1}{4x}$ and $(\frac{f(x)}{g(x)})' = \frac{f'(x) g(x) - f(x) g'(x)}{g(x)^2} = \frac{2 x^2 -1 - (x+2)*(2x)}{(2x^2 - 1)^2} = \frac{-2x -1}{(2x^2 - 1)^2}$
	    \end{solution}
    \end{parts}
