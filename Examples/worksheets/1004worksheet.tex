	\question
Find $\frac{dy}{dx}$ for each equation.
        \begin{parts}
        \part $(x-y)^2=x+y-1$
\begin{solution}[2cm]
	\[\frac d{dx}\Big((x-y)^2\Big)=\frac d{dx}(x+y-1)\]
            \[2(x-y)\cdot\left(1-\frac{dy}{dx}\right)=1+\frac{dy}{dx}\]
            \[2x-2x\frac{dx}{dy}-2y+2y\frac{dy}{dx}=1\frac{dy}{dx}\]
            \[\frac{dy}{dx}=\frac{2y-2x+1}{2y-2x-1}\]
\end{solution}
        \part $\cos^2x+\cos^2y=\cos(2x+2y)$
\begin{solution}[2cm]
	\[\frac{d}{dx}(\cos^2x+\cos^2y)=\frac{d}{dx}\cos(2x+2y)\]
            \[2\cos x\cdot\sin x+2\cos y\cdot\sin y\cdot\frac{dy}{dx}=\sin(2x+2y)\cdot(2+2\frac{dy}{dx})\]
            \[2\cos x\sin x+2\cos y\sin y\frac{dy}{dx}=2\sin(2x+2y)+2\frac{dy}{dx}\sin(2x+2y)\]
        \[\frac{dy}{dx}=\frac{\cos x\sin x-\sin(2x+2y)}{\sin(2x+2y)-\cos y\sin y}\]
\end{solution}
        \end{parts}

	\question
Find the equation of the tangent line to the curve at the given point
\begin{parts}
		    \part $x^2+xy+y^2=3$ at $(1,1)$
\begin{solution}[3cm]
\[\frac d{dx}(x^2+xy+y^2)=\frac d{dx}3\]
        \[2x+x\frac{dy}{dx}+1y+2y\frac{dy}{dx}=0\]
        We are only interested in the slope at a certain point, so instead of solving for $\frac{dy}{dx}$ now, it's easier to substitute the point $(1,1)$ in now, and solve for the slope. We'll change $\frac{dy}{dx}$ to $m$ to signify that it's the slope at a certain point.
        
        \[2(1)+1m+1(1)+2(1)m=0\]
        \[3+3m=0\]
        \[m=-1\]
        
        So the equation of the tangent line is \[y-1=-1(x-1)\]
\end{solution}
			\part $y = \frac{2}{\sin x + \cos x}$ at $(0,1)$
\begin{solution}[3cm]
	$y' = \frac{-2}{(\sin x + \cos x)^2} \cdot (\cos x - \sin x)$. Thus $y' (0) = \frac{-2}{1} \cdot 1 = -2$.
    
    The equation of the tangent line is then $y-1 = -2 x$.
\end{solution}
