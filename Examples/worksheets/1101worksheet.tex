		We are given information about $f$ below. What could $f$ possibly be (there might be several answers)?	
        \begin{parts}
            \part $f'(x)=x^{4/5}$
		\begin{solution}[\stretch{1}]
		    $\frac{5}{9} x^{9/5} + C$
	    \end{solution}
            \part $f'(x)=x(\sqrt[3]x+\sqrt[4]x)$
\begin{solution}[\stretch{1}]
	$\frac{3 x^{7/3}}{7} + \frac{4 x^{9/4}}{9} + C$
	    \end{solution}
            \part $f'(x)=\sqrt{x}(6+5x)$, $f(1)=10$
\begin{solution}[\stretch{1}]
	$f(x)=4x^{\frac32}+2x^{\frac52}+4$
	    \end{solution}
            \part $f''(x)=24x^2+2x+10$, $f(1)=5$, $f'(1)=-3$
\begin{solution}[\stretch{1}]
	$f(x)=2x^4+\frac13x^3+5x^2-22x+\frac{59}3$
	    \end{solution}
            \part $f''(\theta)=\sin\theta+\cos\theta$, $f(0)=3$, $f'(0)=4$
\begin{solution}[\stretch{1}]
	$f(\theta)=-\sin\theta-\cos\theta+5\theta+4$
	    \end{solution}
